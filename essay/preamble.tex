\newcommand\OnRight[1]{%
  \unskip           % (1)
  \hfill            % (2)
  \penalty100\relax % (3)
  \hspace*{0.5em}%    (4)
  \hspace*{\fill}%    (5)
  \mbox{#1}%          (6)
}

    }%
    \ifthenelse{\equal{#1}{}}{%
        \questionn*{\questionargs}
    }{%
        \renewcommand{\thequestionn}{\questiontext{#1}}
        \questionn{\questionargs}
    }%
